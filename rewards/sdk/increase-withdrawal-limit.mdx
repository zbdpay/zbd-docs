---
title: "Increase Withdrawal Limit"
description: "Programmatically increase a user's withdrawal limit via the REST API."
api: "POST https://api.zbdpay.com/api/v1/rewards/user/{rewardsUserId}/withdrawal-limit"
---

Increase the withdrawal limit for a specific user in your Rewards App. This endpoint allows you to raise the maximum amount a user can withdraw from their rewards balance.

## Configuration

### Header Parameters

<ParamField required header="Authorization" type="string">
  Bearer token for authentication (admin JWT required)

  Format: `Bearer {JWT_TOKEN}`
</ParamField>

<ParamField required header="z-client" type="string" initialValue="developer-dashboard">
  Client identifier (use <code>"developer-dashboard"</code>)
</ParamField>

<ParamField header="x-credential-identifier" type="string">
  User ID extracted from JWT (automatically set)
</ParamField>

### Path Parameters

<ParamField required path="rewardsUserId" type="string">
  The user's rewards ID (UUID format)
</ParamField>

### Body Parameters

<ParamField required body="amount" type="number">
  Amount to increase the withdrawal limit by. Must be a positive number greater than 0.
</ParamField>

<ParamField required body="currency" type="string">
  Currency type for the amount

  Options: `MSATS` or `POINT`
</ParamField>

<ResponseExample>
```json 200 - Success
{
  "success": true,
  "message": "Success."
}
```
</ResponseExample>

### Response Status Codes

| Code | Description |
|------|-------------|
| `200` | Limit added successfully |
| `400` | Bad request - invalid parameters or negative amount |
| `401` | Unauthorized - authentication required |
| `403` | Forbidden - admin access required |
| `404` | User not found |
| `500` | Internal server error |

```javascript Node.js
const rewardsUserId = 'USER_REWARDS_ID';
const jwtToken = 'YOUR_JWT_TOKEN';

async function increaseWithdrawalLimit(rewardsUserId, amount, currency) {
  const response = await fetch(
    `https://api.zbdpay.com/api/v1/rewards/user/${rewardsUserId}/withdrawal-limit`,
    {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${jwtToken}`,

      },
      body: JSON.stringify({

  if (data.success) {
    console.log('Withdrawal limit increased successfully');
    return data;
  } else {
    throw new Error(`Failed: ${data.message}`);
  }
}
```

```python Python
import requests
import os

jwt_token = os.getenv('JWT_TOKEN')
rewards_user_id = "USER_REWARDS_ID"

def increase_withdrawal_limit(rewards_user_id, amount, currency):
    """Increase a user's withdrawal limit"""

    url = f"https://api.zbdpay.com/api/v1/rewards/user/{rewards_user_id}/withdrawal-limit"
    headers = {
        "Authorization": f"Bearer {jwt_token}",
        "z-client": "developer-dashboard",
        "Content-Type": "application/json"
    }
    payload = {
        "amount": amount,
        "currency": currency
    }

    response = requests.post(url, headers=headers, json=payload)
    data = response.json()

    if data["success"]:
        print("Withdrawal limit increased successfully")
        return data
    else:
        raise Exception(f"Failed: {data['message']}")

# Increase limit by 1000 MSATS
increase_withdrawal_limit("b28e0306-2c06-4092-8d56-a1623d6b97fb", 1000, "MSATS")
```

```go Go
package main

import (
    "bytes"
    "encoding/json"
    "fmt"
    "io"
    "net/http"
    "os"
)

type IncreaseRequest struct {
    Amount   int    `json:"amount"`
    Currency string `json:"currency"`
}

type Response struct {
    Success bool   `json:"success"`
    Message string `json:"message"`
}

func increaseWithdrawalLimit(rewardsUserId string, amount int, currency string) (*Response, error) {
    jwtToken := os.Getenv("JWT_TOKEN")
    url := fmt.Sprintf("https://api.zbdpay.com/api/v1/rewards/user/%s/withdrawal-limit", rewardsUserId)

    requestBody := IncreaseRequest{
        Amount:   amount,
        Currency: currency,
    }

    jsonData, err := json.Marshal(requestBody)
    if err != nil {
        return nil, err
    }

    req, err := http.NewRequest("POST", url, bytes.NewBuffer(jsonData))
    if err != nil {
        return nil, err
    }

    req.Header.Set("Authorization", "Bearer "+jwtToken)
    req.Header.Set("z-client", "developer-dashboard")
    req.Header.Set("Content-Type", "application/json")

    client := &http.Client{}
    resp, err := client.Do(req)
    if err != nil {
        return nil, err
    }
    defer resp.Body.Close()

    body, err := io.ReadAll(resp.Body)
    if err != nil {
        return nil, err
    }

    var response Response
    err = json.Unmarshal(body, &response)
    if err != nil {
        return nil, err
    }

    if response.Success {
        fmt.Println("Withdrawal limit increased successfully")
    }

    return &response, nil
}

func main() {
    // Increase limit by 1000 MSATS
    response, err := increaseWithdrawalLimit("b28e0306-2c06-4092-8d56-a1623d6b97fb", 1000, "MSATS")
    if err != nil {
        fmt.Printf("Error: %v\n", err)
        return
    }
    fmt.Printf("Response: %+v\n", response)
}
```
